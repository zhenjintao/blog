vue2中的响应式（依赖收集&触发） ​

getter时收集watcher，setter时出发watcher

html
<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Simple Vue Two - Way Binding with Observer Pattern</title>
</head>

<body>
    <div id="app">
        <input type="text" v-model="message">
        <p>{{ message }}</p>
    </div>
    <script>
        // 发布者类，管理订阅者并在数据变化时通知它们
        class Dep {
            constructor() {
                // 存储订阅者的数组
                this.subscribers = [];
            }

            // 添加订阅者
            addSubscriber(subscriber) {
                this.subscribers.push(subscriber);
            }

            // 通知所有订阅者数据已更新
            notify() {
                this.subscribers.forEach(subscriber => subscriber.update());
            }
        }

        // 观察者类，当数据变化时执行更新操作
        class Watcher {
            constructor(vm, key, updateFn) {
                this.vm = vm;
                this.key = key;
                this.updateFn = updateFn;
                // 触发数据的 getter，将当前 Watcher 添加到 Dep 中
                Dep.target = this;
                this.vm[this.key];
                Dep.target = null;
            }

            // 更新视图
            update() {
                this.updateFn(this.vm[this.key]);
            }
        }

        // 对数据对象进行数据劫持，使其具有响应式特性
        function observe(data) {
            if (typeof data !== 'object' || data === null) {
                return;
            }
            for (let key in data) {
                let value = data[key];
                const dep = new Dep();
                Object.defineProperty(data, key, {
                    get() {
                        // 如果当前有正在创建的 Watcher，将其添加到 Dep 的订阅者列表中
                        debugger
                        if (Dep.target) {
                            dep.addSubscriber(Dep.target);
                        }
                        return value;
                    },
                    set(newValue) {
                        if (value !== newValue) {
                            value = newValue;
                            // 数据变化时通知所有订阅者
                            dep.notify();
                        }
                    }
                });
                // 递归处理嵌套对象
                observe(value);
            }
        }

        // 模拟 Vue 实例
        class Vue {
            constructor(options) {
                this.$options = options;
                this.$data = options.data;
                // 对数据进行数据劫持
                observe(this.$data);
                // 代理数据，使得可以通过 vm.key 访问 vm.$data.key
                this.proxyData(this.$data);
                // 编译模板
                this.compile(options.el);
            }

            // 代理数据，方便直接通过实例访问数据
            proxyData(data) {
                Object.keys(data).forEach(key => {
                    Object.defineProperty(this, key, {
                        get() {
                            debugger
                            return data[key];
                        },
                        set(newValue) {
                            data[key] = newValue;
                        }
                    });
                });
            }

            // 编译模板，处理 v-model 和插值表达式
            compile(el) {
                const element = document.querySelector(el);
                const input = element.querySelector('input[v-model]');
                const p = element.querySelector('p');
                const key = input.getAttribute('v-model');

                // 创建 Watcher 来监听数据变化并更新段落文本
                new Watcher(this, key, (newValue) => {
                    p.textContent = newValue;
                });

                // 监听输入框的输入事件，更新数据
                input.addEventListener('input', (e) => {
                    this[key] = e.target.value;
                });

                // 初始化段落文本
                p.textContent = this[key];
            }
        }

        // 创建 Vue 实例
        const app = new Vue({
            el: '#app',
            data: {
                message: 'Hello, Observer Pattern!'
            }
        });
    </script>
</body>

</html>
